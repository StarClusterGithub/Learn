/*
=================================================================
小明有5本书，要借给3个同学，若每人每次只能借一本，
试编写程序，计算有多少种不同的借法？并输出这些借法。
=================================================================
*/
#include<stdio.h>
int main(void)
{
	int a,b,c,number=1;
	for(a=65;a<=69;a++)
		for(b=65;b<=69;b++)
			for(c=65;c<=69;c++)
				if(a!=b&&a!=c&&b!=c)
					printf("%3d:  %-3c%-3c%-3c\n",number++,a,b,c);
	return 0;
}
