//奇阶幻方函数
void unevennumber(int m,int n,int i,int j,int *luoshu)
{
	//生成奇阶幻方
	for(m=1,i=0,j=(n-1)/2;m<=n*n;++m,--i,++j)
	{
		if(i<0)
			i+=n;
		if(j==n)
			j-=n;
		if(luoshu[i*n+j]==0)
			luoshu[i*n+j]=m;
		else
		{
			++i;--j;
			if(i==n)i-=n;
			if(j<0)j+=n;
			++i;
			if(i<0)i+=n;
			if(j==n)j-=n;
			luoshu[i*n+j]=m;
		}
	}
	//输出
	for(i=0;i<n;++i)
	{
		printf("\n");
		for(j=0;j<n;++j)
		printf("%4d",luoshu[i*n+j]);
		printf("\n");
	}
	printf("\n");
}

//单偶阶幻方函数
void evennumber(int m,int n,int i,int j,int *luoshu)
{
	int k,l,t;
	//填数
	for(m=1,i=0,j=(n/2-1)/2,l=2;m<=n*n;++m,--i,++j)
	{
		//第二象限
		if(1<=m&&m<=n*n/4)
		{
			if(i<0)i+=(n/2);
			if(j==n/2)j-=(n/2);
			if(luoshu[i*n+j]==0)
				luoshu[i*n+j]=m;
			else
			{
				++i;--j;
				if(i==n/2)i-=(n/2);
				if(j<0)j+=(n/2);
				++i;
				if(i<0)i+=(n/2);
				if(j==n/2)j-=(n/2);
				luoshu[i*n+j]=m;
			}
		}
		//第四象限
		if(n*n/4<m&&m<=n*n/2)
		{
			if(l==2)
			{
				l++;
				i=n/2;
				j=(n/2+n-1)/2;
			}
			if(i<n/2)i+=(n/2);
			if(j==n)j-=(n/2);
			if(luoshu[i*n+j]==0)
				luoshu[i*n+j]=m;
			else
			{
				++i;--j;
				if(i==n)i-=(n/2);
				if(j<n/2)j+=(n/2);
				++i;
				if(i<n/2)i+=(n/2);
				if(j==n)j-=(n/2);
				luoshu[i*n+j]=m;
			}
		}
		//第一象限
		if(n*n/2<m&&m<=n*n/4*3)
		{
			if(l==3)
			{
				l++;
				i=0;
				j=(n/2+n-1)/2;
			}
			if(i<0)i+=(n/2);
			if(j==n)j-=(n/2);
			if(luoshu[i*n+j]==0)
				luoshu[i*n+j]=m;
			else
			{
				++i;--j;
				if(i==n/2)i-=(n/2);
				if(j<n/2)j+=(n/2);
				++i;
				if(i<0)i+=(n/2);
				if(j==n)j-=(n/2);
				luoshu[i*n+j]=m;
			}
		}
		//第三象限
		if(n*n/4*3<m&&m<=n*n)
		{
			if(l==4)
			{
				l++;
				i=n/2;
				j=(n/2-1)/2;
			}
			if(i<n/2)i+=(n/2);
			if(j==n/2)j-=(n/2);
			if(luoshu[i*n+j]==0)
				luoshu[i*n+j]=m;
			else
			{
				++i;--j;
				if(i==n)i-=(n/2);
				if(j<0)j+=(n/2);
				++i;
				if(i<n/2)i+=(n/2);
				if(j==n/2)j-=(n/2);
				luoshu[i*n+j]=m;
			}
		}
	}
	//换位
	for(k=0;k<n/4;++k)
	{
		for(i=0,j=(n/2+n-1)/2;i<n/2;++i)
		{
			if(k>=n/4-1)break;
			t=luoshu[i*n+j-k];
			luoshu[i*n+j-k]=luoshu[(i+n/2)*n+j-k];
			luoshu[(i+n/2)*n+j-k]=t;
		}
		for(i=0,j=0;i<n/2;++i)
		{
			if(i==(n/2-1)/2)
			{
				t=luoshu[i*n+(n/2-1)/2+k];
				luoshu[i*n+(n/2-1)/2+k]=luoshu[(i+n/2)*n+(n/2-1)/2+k];
				luoshu[(i+n/2)*n+(n/2-1)/2+k]=t;
			}
			else
			{
				t=luoshu[i*n+j+k];
				luoshu[i*n+j+k]=luoshu[(i+n/2)*n+j+k];
				luoshu[(i+n/2)*n+j+k]=t;
			}
		}

	}
	//输出
	for(i=0;i<n;++i)
	{
		printf("\n");
		for(j=0;j<n;++j)
			printf("%5d",luoshu[i*n+j]);
		printf("\n");
	}
	printf("\n");
}

//双偶阶幻方函数
void evennumbers(int m,int n,int i,int j,int *luoshu)
{
	//填数
	for(m=1,i=0;i<n;++i)
		for(j=0;j<n;++m,++j)
			luoshu[i*n+j]=m;
	//将对角线上的数替换为互补数
	for(i=0;i<n;i+=4)
		for(j=0;j<n;j+=4)
			luoshu[i*n+j]=n*n+1-luoshu[i*n+j];
	for(i=1;i<n;i+=4)
		for(j=1;j<n;j+=4)
			luoshu[i*n+j]=n*n+1-luoshu[i*n+j];
	for(i=2;i<n;i+=4)
		for(j=2;j<n;j+=4)
			luoshu[i*n+j]=n*n+1-luoshu[i*n+j];
	for(i=3;i<n;i+=4)
		for(j=3;j<n;j+=4)
			luoshu[i*n+j]=n*n+1-luoshu[i*n+j];
	for(i=0;i<n;i+=4)
		for(j=3;j<n;j+=4)
			luoshu[i*n+j]=n*n+1-luoshu[i*n+j];
	for(i=1;i<n;i+=4)
		for(j=2;j<n;j+=4)
			luoshu[i*n+j]=n*n+1-luoshu[i*n+j];
	for(i=2;i<n;i+=4)
		for(j=1;j<n;j+=4)
			luoshu[i*n+j]=n*n+1-luoshu[i*n+j];
	for(i=3;i<n;i+=4)
		for(j=0;j<n;j+=4)
			luoshu[i*n+j]=n*n+1-luoshu[i*n+j];
	//输出
	for(i=0;i<n;++i)
	{
		printf("\n");
		for(j=0;j<n;++j)
			printf("%5d",luoshu[i*n+j]);
		printf("\n");
	}
	printf("\n");
}
