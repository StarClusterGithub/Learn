/*【河图洛书】幻方，即中华古文明中的河图洛书。
河图与洛书是中国古代流传下来的两幅神秘图案，历来被认为是河洛文化的滥觞，中华文明的源头，被誉为"宇宙魔方"。
相传，上古伏羲氏时，洛阳东北孟津县境内的黄河中浮出龙马，背负"河图"，献给伏羲。伏羲依此而演成八卦，后为《周易》来源。
又相传，大禹时，洛阳境内洛河中浮出神龟，背驮"洛书"，献给大禹。大禹依此治水成功，遂划天下为九州。
又依此定九章大法，治理社会，流传下来收入《尚书》中，名《洪范》。《易・系辞上》说："河出图，洛出书，圣人则之"，就是指这两件事。
河图上，排列成数阵的黑点和白点，蕴藏着无穷的奥秘；洛书上，纵、横、斜三条线上的三个数字，其和皆等于15，十分奇妙。
对此，中外学者作了长期的探索研究，认为这是中国先民心灵思维的结晶，是中国古代文明的第一个里程碑。
《周易》和《洪范》两书，在中华文化发展史上有着重要的地位，在哲学、政治学、军事学、伦理学、美学、文学诸领域产生了深远影响。
作为中国历史文化渊源的河图洛书，功不可没。*/
#include<stdio.h>
#include<stdlib.h>
//100以内的单偶数阶幻方
int main()
{
int n,m,i,j,k,l,t,luoshu[100][100];
printf("请输入幻方的阶数：");
scanf("%d",&n);
system("cls");
for(i=0;i<100;++i)
  for(j=0;j<100;++j)
	  luoshu[i][j]=0;
for(m=1,i=0,j=(n/2-1)/2,l=2;m<=n*n;++m,--i,++j)
  {
//第二象限
	if(1<=m&&m<=n*n/4)
	{
		if(i<0)i+=(n/2);
		if(j==n/2)j-=(n/2);
		if(luoshu[i][j]==0)
		  luoshu[i][j]=m;
		else
		{
		  ++i;--j;
		  if(i==n/2)i-=(n/2);
		  if(j<0)j+=(n/2);
		  ++i;
		  if(i<0)i+=(n/2);
		  if(j==n/2)j-=(n/2);
		  luoshu[i][j]=m;
		}
	}
//第四象限
	if(n*n/4<m&&m<=n*n/2)
	{
		if(l==2)
		{
			l++;
			i=n/2;
			j=(n/2+n-1)/2;
		}
		if(i<n/2)i+=(n/2);
		if(j==n)j-=(n/2);
		if(luoshu[i][j]==0)
		  luoshu[i][j]=m;
		else
		{
		  ++i;--j;
		  if(i==n)i-=(n/2);
		  if(j<n/2)j+=(n/2);
		  ++i;
		  if(i<n/2)i+=(n/2);
		  if(j==n)j-=(n/2);
		  luoshu[i][j]=m;
		}
	}
//第一象限
  	if(n*n/2<m&&m<=n*n/4*3)
	{
		if(l==3)
		{
			l++;
			i=0;
			j=(n/2+n-1)/2;
		}
		if(i<0)i+=(n/2);
		if(j==n)j-=(n/2);
		if(luoshu[i][j]==0)
		  luoshu[i][j]=m;
		else
		{
		  ++i;--j;
		  if(i==n/2)i-=(n/2);
		  if(j<n/2)j+=(n/2);
		  ++i;
		  if(i<0)i+=(n/2);
		  if(j==n)j-=(n/2);
		  luoshu[i][j]=m;
		}
	}
//第三象限
	if(n*n/4*3<m&&m<=n*n)
	{
		if(l==4)
		{
			l++;
			i=n/2;
			j=(n/2-1)/2;
		}
		if(i<n/2)i+=(n/2);
		if(j==n/2)j-=(n/2);
		if(luoshu[i][j]==0)
		  luoshu[i][j]=m;
		else
		{
		  ++i;--j;
		  if(i==n)i-=(n/2);
		  if(j<0)j+=(n/2);
		  ++i;
		  if(i<n/2)i+=(n/2);
		  if(j==n/2)j-=(n/2);
		  luoshu[i][j]=m;
		}
	}
  }
for(k=0;k<n/4;++k)
  {
	for(i=0,j=(n/2+n-1)/2;i<n/2;++i)
	{
		if(k>=n/4-1)break;
		t=luoshu[i][j-k];
		luoshu[i][j-k]=luoshu[i+n/2][j-k];
		luoshu[i+n/2][j-k]=t;
	}
	for(i=0,j=0;i<n/2;++i)
	{
		if(i==(n/2-1)/2)
		  {
			t=luoshu[i][(n/2-1)/2+k];
			luoshu[i][(n/2-1)/2+k]=luoshu[i+n/2][(n/2-1)/2+k];
			luoshu[i+n/2][(n/2-1)/2+k]=t;
		  }
		else
	  	  {
			t=luoshu[i][j+k];
			luoshu[i][j+k]=luoshu[i+n/2][j+k];
			luoshu[i+n/2][j+k]=t;
	      }
	}

  }
for(i=0;i<n;++i)
  {
	printf("\n");
	for(j=0;j<n;++j)
		printf("%5d",luoshu[i][j]);
	printf("\n");
  }
return 0;
}
